// Definimos los objetos instructor y alumno
Objeto instructor
    nombre = "Franco"
    edad = 25
Fin Objeto

Objeto alumno
    nombre = "Juan"
    curso = "FullStack"
Fin Objeto

// Definimos la función getNombre que retorna el nombre del objeto actual
Función getNombre()
    Retornar this.nombre
Fin Función

// Solución usando bind()
// Creamos dos funciones usando bind() y establecemos el valor de this en los objetos correspondientes
Función getNombreInstructor = getNombre.bind(instructor)
Función getNombreAlumno = getNombre.bind(alumno)

// Imprimimos los resultados
Imprimir getNombreInstructor() // Salida: 'Franco'
Imprimir getNombreAlumno() // Salida: 'Juan'
