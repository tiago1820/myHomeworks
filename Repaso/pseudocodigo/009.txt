Clase Queue
    // Constructor de la clase Queue
    Función Constructor()
        cola <- Lista vacía // Creamos una lista vacía para almacenar los elementos de la cola
    Fin Función

    // Método para agregar un valor a la cola (enqueue)
    Función enqueue(valor)
        cola.agregarAlFinal(valor) // Agregamos el valor al final de la lista (el último en ingresar a la cola)
    Fin Función

    // Método para remover un valor de la cola (dequeue)
    Función dequeue()
        Si cola está vacía // Verificamos si la cola está vacía
            Retornar indefinido // Si está vacía, retornamos undefined
        Fin Si
        valorRemovido <- cola.primerElemento() // Obtenemos el primer elemento de la lista (el primero en ingresar a la cola)
        cola.removerPrimerElemento() // Removemos el primer elemento de la lista (el primero en ingresar a la cola)
        Retornar valorRemovido // Retornamos el valor que fue removido de la cola
    Fin Función

    // Método para obtener el tamaño de la cola (size)
    Función size()
        Retornar cola.tamaño() // Retornamos el tamaño (cantidad de elementos) de la lista que representa la cola
    Fin Función
Fin Clase
